# build docker
docker build -t containername 
docker build -f Dockerfile -t containername .

# push image to hub
docker login "https://docker.hub.org/url"
docker push https://docker.hub.org/url/$VERSION_TAG

# docker run
docker run -it containername bash

# docker-compose
## run container
docker-compose up -d

## stop container (keep changes in container)
docker-compose stop 

## reset container
docker-compose down

# delete unused images
docker images 
docker rmi 'image-id'

# delete exited containers
docker rm $(docker ps -qa --no-trunc --filter "status=exited")

# get infos about container status 
docker ps -a 

# get info about docker 
docker info
